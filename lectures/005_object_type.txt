Object Type
    จัดเก็บข้อมูลในรูปแบบของ key:value
    เช่น
        const person: object = {
            name: "bike",
            age: 30
        }

    และเราสามารถกำหนดว่า object ดังกล่าวจะต้องมี properties อะไรบ้าง
    เช่น
        const person: {
            name: string,
            age: number
        } = {
            name: "bike",
            age: 30
        }
